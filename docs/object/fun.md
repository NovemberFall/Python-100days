## def Keyword

```py
def say_hello():
    print("hello")
    print('are')
    print('you')
say_hello()
hello
are
you




def say_hello(name):
    print(f'Hello {name}')
say_hello('Jose')
Hello Jose





def say_hello(name='Default'):
    print(f'Hello {name}')
say_hello()
Hello Default






def add_num(num1, num2):
    return num1 + num2

result = add_num(10, 20)
result
30




def print_result(a,b):
    print(a+b)

def return_result(a,b):
    return a+b

result = print_result(10,20)

result

return_result(10,20)

    print(a+b)
```
